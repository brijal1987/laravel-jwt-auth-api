# laravel-jwt-auth-api
API with JWT 

# Clone Directory
git clone https://github.com/brijal1987/laravel-jwt-auth-api.git

# Install Libraries
composer install

# Clone repo and composer update
Setup Database for Laravel
DB_CONNECTION=mysql
DB_HOST=localhost
DB_PORT=3306
DB_DATABASE=dbname
DB_USERNAME=username
DB_PASSWORD=password

# Run Migration
php artisan migrate

# Start Server
php artisan serve
